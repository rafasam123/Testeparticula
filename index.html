  createCanvas(400, 400);
  angleMode(DEGREES);
}

function draw() {
  background(220);
  translate(width / 2, height / 2);
  rotate(angle);
  rectMode(CENTER);
  rect(0, 0, 150, 50);
  angle += 1;
}
